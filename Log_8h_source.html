44 /* Assumes all types passed are valid as it is an internal function */
45 template<typename Ty>
46 std::string ProcessArg(Ty&& arg)
47 {
48 /* Checks if the argument type is a pointer */
49 if constexpr(std::is_pointer_v<std::remove_cvref_t<Ty>>)
50 {
51 /* If the pointer is valid forwards the derefenced arg */
52 if (arg != nullptr)
53 return ProcessArg(*arg);
54
55 /* Else prints a message about a nullptr of type Ty */
56 std::ostringstream os{};
57 os << "Nullptr of type: [" << typeid(Ty).name() << ']';
58 return std::move(os).str();
59 }
60
61 /* Custom log function has highest precedence */
62 else if constexpr (TypeHasLogFunction<Ty>)
63 {
64 return arg.LogStr();
65 }
66
67 /* Checks for standard logging (std::ostream& << Ty) */
68 else if constexpr (StandardLogable<Ty>)
69 {
70 std::ostringstream os{};
71 os << arg;
72
73 return std::move(os).str();
74 }
75
76 /* Else returns an error */
77 else
78 {
79 static_assert(false, "Invalid type passed to Util::Internal::ProcessArg(), It is recommended not to use internal functions");
80 }
81 }
